\section{算法}

\begin{algorithm}
    \renewcommand{\algorithmicrequire}{\textbf{Input:}}
    \renewcommand{\algorithmicensure}{\textbf{Output:}}
    \caption{斐波那契数列算法}
    \label{alg:fibonacci}
    \begin{algorithmic}[1]
        \REQUIRE  斐波那契数列的长度$n$
        \ENSURE 斐波那契数列的前$n$项
        \STATE Initialize $F[0] \gets 0$, $F[1] \gets 1$
        \FOR{$i = 2$ to $n$}
        \STATE $F[i] \gets F[i-1] + F[i-2]$
        \ENDFOR
        \STATE \textbf{return} 斐波那契数列 $F[0], F[1], \ldots, F[n-1]$
    \end{algorithmic}
\end{algorithm}
